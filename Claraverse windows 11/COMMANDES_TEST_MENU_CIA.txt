### Scénario 3: Test Calcul Score

// Créer une fonction de test
function testScoreCalculation() {
  const table = document.querySelector('table');
  
  // Avant calcul
  console.log('📊 AVANT CALCUL:');
  console.log('Score:', table.dataset.totalScore || 'Non calculé');
  
  // Calculer
  window.ClaraverseCIAMenu.calculateScores(table);
  
  // Après calcul
  console.log('📊 APRÈS CALCUL:');
  console.log('Score:', table.dataset.totalScore);
  console.log('Questions:', table.dataset.questionCount);
  console.log('Pourcentage:', 
    (parseInt(table.dataset.totalScore) / parseInt(table.dataset.questionCount) * 100).toFixed(1) + '%'
  );
}

testScoreCalculation();

## 🚀 COMMANDES AVANCÉES

### Exporter les résultats
```javascript
function exportResults() {
  const tables = document.querySelectorAll('table');
  const results = [];
  
  tables.forEach((table, i) => {
    if (window.ClaraverseCIAMenu.isCIATable(table)) {
      results.push({
        table: i + 1,
        score: table.dataset.totalScore || 0,
        questions: table.dataset.questionCount || 0
      });
    }
  });
  
  console.table(results);
  return results;
}

exportResults();
```

### Statistiques globales
```javascript
function getGlobalStats() {
  const tables = Array.from(document.querySelectorAll('table'))
    .filter(t => window.ClaraverseCIAMenu.isCIATable(t));
  
  const totalScore = tables.reduce((sum, t) => 
    sum + parseInt(t.dataset.totalScore || 0), 0);
  
  const totalQuestions = tables.reduce((sum, t) => 
    sum + parseInt(t.dataset.questionCount || 0), 0);
  
  const percentage = (totalScore / totalQuestions * 100).toFixed(1);
  
  console.log('📊 STATISTIQUES GLOBALES:');
  console.log(`Tables CIA: ${tables.length}`);
  console.log(`Score total: ${totalScore}/${totalQuestions}`);
  console.log(`Pourcentage: ${percentage}%`);
  
  return { tables: tables.length, totalScore, totalQuestions, percentage };
}

getGlobalStats();
```
