document.querySelectorAll('table').forEach((table, i) => {
  console.log(`\n📊 Table ${i+1}:`);
  const rows = table.querySelectorAll('tbody tr');
  rows.forEach((row, j) => {
    const cells = row.querySelectorAll('td');
    cells.forEach((cell, k) => {
      const header = table.querySelectorAll('th')[k];
      if (header && header.textContent.toLowerCase().includes('remarques')) {
        console.log(`  Ligne ${j+1}:`, cell.textContent.substring(0, 50));
        console.log(`    Display: ${cell.style.display}`);
        console.log(`    Merged: ${cell.getAttribute('data-merged')}`);
        console.log(`    Rowspan: ${cell.getAttribute('rowspan')}`);
      }
    });
  });
});

✅ VALIDATION RAPIDE

Test complet en une commande
──────────────────────────────────────────────────────────────
(function() {
  console.log('🧪 TEST DE VALIDATION TÂCHE 7');
  console.log('═══════════════════════════════════════════════');
  
  const tests = {
    'Script chargé': !!window.ClaraverseTableProcessor,
    'Tables détectées': document.querySelectorAll('table').length > 0,
    'Cellules fusionnées': document.querySelectorAll('[data-merged="true"]').length > 0,
    'Rowspan présent': document.querySelectorAll('[rowspan]').length > 0,
    'Colonnes masquées': document.querySelectorAll('[data-hidden="true"]').length > 0,
    'localStorage actif': !!localStorage.getItem('claraverse_tables_data')
  };
  
  let passed = 0;
  let failed = 0;
  
  for (const [test, result] of Object.entries(tests)) {
    const status = result ? '✅' : '❌';
    console.log(`${status} ${test}: ${result}`);
    result ? passed++ : failed++;
  }
  
  console.log('═══════════════════════════════════════════════');
  console.log(`Résultat: ${passed}/${Object.keys(tests).length} tests réussis`);
  
  if (failed === 0) {
    console.log('🎉 TOUS LES TESTS SONT PASSÉS !');
  } else {
    console.log(`⚠️ ${failed} test(s) échoué(s)`);
  }
  
  console.log('═══════════════════════════════════════════════');
})();
