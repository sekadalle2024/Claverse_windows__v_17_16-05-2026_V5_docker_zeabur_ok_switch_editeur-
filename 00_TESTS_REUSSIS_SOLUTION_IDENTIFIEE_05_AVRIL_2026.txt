```python
def generer_etats_controle_pour_export(results: Dict[str, Any]) -> List[Dict[str, Any]]:
    """
    Génère les 16 états de contrôle à partir des résultats des états financiers
    UTILISE LES 16 ÉTATS ACTUALISÉS
    """
    from etats_controle_exhaustifs import generer_controles_exhaustifs
    
    # Extraire les données
    balance_n = results.get('balance_n', [])
    balance_n1 = results.get('balance_n1', [])
    bilan_actif = results.get('bilan_actif', [])
    bilan_passif = results.get('bilan_passif', [])
    compte_resultat = results.get('compte_resultat', [])
    
    # Générer les contrôles exhaustifs
    controles_n, controles_n1, totaux_n, totaux_n1 = generer_controles_exhaustifs(
        balance_n=balance_n,
        balance_n1=balance_n1,
        bilan_actif=bilan_actif,
        bilan_passif=bilan_passif,
        compte_resultat=compte_resultat
    )
    
    # Convertir en format Excel (16 états)
    etats_controle = []
    
    # 1. Statistiques de Couverture (N)
    stats = controles_n.get('statistiques', {})
    etats_controle.append({
        'titre': '1. Statistiques de Couverture (Exercice N)',
        'postes': [
            {'ref': 'CA', 'libelle': 'Comptes dans la balance', 
             'montant_n': stats.get('total_comptes_balance', 0), 'montant_n1': 0},
            {'ref': 'CB', 'libelle': 'Comptes intégrés', 
             'montant_n': stats.get('comptes_integres', 0), 'montant_n1': 0},
            {'ref': 'CC', 'libelle': 'Taux de couverture (%)', 
             'montant_n': stats.get('taux_couverture', 0), 'montant_n1': 0},
        ]
    })
    
    # 2-16. Autres états...
    
    return etats_controle
```
