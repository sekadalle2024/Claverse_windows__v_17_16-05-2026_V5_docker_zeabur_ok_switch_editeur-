🎯 OBJECTIF
─────────────────────────────────────────────────────────────────────────────

Commandes rapides pour tester et déboguer la nouvelle nomenclature des fichiers
exportés via "Export template Excel".

🔧 COMMANDES CONSOLE (F12)
─────────────────────────────────────────────────────────────────────────────

┌─────────────────────────────────────────────────────────────────────────┐
│ 1. Vérifier que la fonction existe                                     │
└─────────────────────────────────────────────────────────────────────────┘

console.log(typeof window.contextualMenuManager?.extractEtapeMission);
// Devrait afficher: "function"

┌─────────────────────────────────────────────────────────────────────────┐
│ 2. Tester manuellement l'extraction                                    │
└─────────────────────────────────────────────────────────────────────────┘

const tables = document.querySelectorAll('table');
const etape = window.contextualMenuManager?.extractEtapeMission(Array.from(tables));
console.log("Étape trouvée:", etape);

┌─────────────────────────────────────────────────────────────────────────┐
│ 3. Afficher toutes les tables du message                               │
└─────────────────────────────────────────────────────────────────────────┘

const tables = document.querySelectorAll('div.prose table');
console.log(`Nombre de tables: ${tables.length}`);
tables.forEach((table, i) => {
  console.log(`Table ${i + 1}:`, {
    rows: table.querySelectorAll('tr').length,
    cols: table.querySelectorAll('tr')[0]?.querySelectorAll('td, th').length || 0
  });
});

┌─────────────────────────────────────────────────────────────────────────┐
│ 4. Analyser une table spécifique                                       │
└─────────────────────────────────────────────────────────────────────────┘

const table = document.querySelectorAll('table')[0]; // Première table
const rows = table.querySelectorAll('tr');
console.log("Contenu de la table:");
rows.forEach((row, i) => {
  const cells = row.querySelectorAll('td, th');
  const rowData = Array.from(cells).map(cell => cell.textContent.trim());
  console.log(`Ligne ${i + 1}:`, rowData);
});

┌─────────────────────────────────────────────────────────────────────────┐
│ 5. Chercher les colonnes "Rubrique" et "Description"                   │
└─────────────────────────────────────────────────────────────────────────┘

const table = document.querySelectorAll('table')[0];
const headerRow = table.querySelector('tr');
const headers = Array.from(headerRow.querySelectorAll('th, td')).map(h => h.textContent.trim());
console.log("En-têtes:", headers);
console.log("Index Rubrique:", headers.findIndex(h => h.toLowerCase().includes('rubrique')));
console.log("Index Description:", headers.findIndex(h => h.toLowerCase().includes('description')));

┌─────────────────────────────────────────────────────────────────────────┐
│ 6. Chercher la ligne "Etape de mission"                                │
└─────────────────────────────────────────────────────────────────────────┘

const table = document.querySelectorAll('table')[0];
const rows = table.querySelectorAll('tr');
rows.forEach((row, i) => {
  const cells = row.querySelectorAll('td, th');
  const firstCell = cells[0]?.textContent.trim().toLowerCase();
  if (firstCell && firstCell.includes('etape') && firstCell.includes('mission')) {
    console.log(`Ligne "Etape de mission" trouvée à l'index ${i}:`, 
                Array.from(cells).map(c => c.textContent.trim()));
  }
});

┌─────────────────────────────────────────────────────────────────────────┐
│ 7. Simuler l'export (sans télécharger)                                 │
└─────────────────────────────────────────────────────────────────────────┘

const tables = document.querySelectorAll('table');
const etape = window.contextualMenuManager?.extractEtapeMission(Array.from(tables));
const timestamp = new Date().toISOString().slice(0, 19).replace(/[:]/g, "-");
let filename;
if (etape) {
  const etapeSanitized = etape.replace(/[^a-zA-Z0-9\s-]/g, '').replace(/\s+/g, '-');
  filename = `E-audit_Etape-mission-${etapeSanitized}_${timestamp}.xlsx`;
} else {
  filename = `E-audit_Etape-mission-X_${timestamp}.xlsx`;
}
console.log("Nom du fichier qui serait généré:", filename);

┌─────────────────────────────────────────────────────────────────────────┐
│ 8. Activer les logs détaillés                                          │
└─────────────────────────────────────────────────────────────────────────┘

// Les logs sont déjà activés par défaut dans extractEtapeMission()
// Pour voir tous les logs, ouvrir la console (F12) avant l'export

┌─────────────────────────────────────────────────────────────────────────┐
│ 9. Tester la sanitisation du nom                                       │
└─────────────────────────────────────────────────────────────────────────┘

const testNames = [
  "Programme de travail",
  "Cartographie des risques!",
  "Lead  sheet",
  "Étape #1",
  "Test@123"
];

testNames.forEach(name => {
  const sanitized = name.replace(/[^a-zA-Z0-9\s-]/g, '').replace(/\s+/g, '-');
  console.log(`"${name}" → "${sanitized}"`);
});

┌─────────────────────────────────────────────────────────────────────────┐
│ 10. Recharger le menu contextuel                                       │
└─────────────────────────────────────────────────────────────────────────┘

// Si le menu ne fonctionne pas après modification
location.reload(); // Recharger la page
// OU
window.contextualMenuManager?.init(); // Réinitialiser le menu
