function getGlobalStats() {
  const tables = Array.from(document.querySelectorAll('table'))
    .filter(t => window.ClaraverseCIAMenu.isCIATable(t));
  
  const totalScore = tables.reduce((sum, t) => 
    sum + parseInt(t.dataset.totalScore || 0), 0);
  
  const totalQuestions = tables.reduce((sum, t) => 
    sum + parseInt(t.dataset.questionCount || 0), 0);
  
  const percentage = (totalScore / totalQuestions * 100).toFixed(1);
  
  console.log('📊 STATISTIQUES:');
  console.log(`Tables: ${tables.length}`);
  console.log(`Score: ${totalScore}/${totalQuestions}`);
  console.log(`Pourcentage: ${percentage}%`);
}

getGlobalStats();
