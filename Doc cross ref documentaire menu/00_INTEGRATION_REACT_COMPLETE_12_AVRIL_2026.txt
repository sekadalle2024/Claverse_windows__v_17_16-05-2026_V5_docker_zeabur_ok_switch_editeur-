   └─ Émet les actions utilisateur

🚀 UTILISATION

Depuis le menu contextuel (menu.js) :

```javascript
// Ouvrir la sidebar avec des documents
document.dispatchEvent(new CustomEvent('xref:open', {
  detail: {
    documents: [
      {
        crossRef: '[I1]',
        document: 'Lead sheet trésorerie',
        client: 'Client A',
        exercice: '2025',
        cycle: 'Trésorerie',
        fileId: 'abc123...',
        fileUrl: 'https://drive.google.com/...'
      }
    ],
    title: 'Documents X-Ref'
  }
}));

// Fermer la sidebar
document.dispatchEvent(new CustomEvent('xref:close'));
```
