🔄 ÉVÉNEMENTS CUSTOMEVENTS

Ouvrir la sidebar :
  document.dispatchEvent(new CustomEvent('xref:open', {
    detail: {
      documents: [...],
      title: 'Titre optionnel'
    }
  }));

Fermer la sidebar :
  document.dispatchEvent(new CustomEvent('xref:close'));

Événements émis :
  • 'xref:closed' - Sidebar fermée
  • 'xref:document:opened' - Document ouvert

📦 STRUCTURE DOCUMENT

interface XRefDocument {
  crossRef: string;           // [I1], [I20], etc.
  document: string;           // Nom du document
  client?: string;            // Client optionnel
  exercice?: string;          // Exercice optionnel
  cycle?: string;             // Cycle comptable optionnel
  dateUpload?: string;        // Date d'upload optionnelle
  fileId?: string;            // ID Google Drive optionnel
  fileUrl?: string;           // URL du fichier optionnelle
}
