🔍 ANALYSE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ BONNE NOUVELLE : La syntaxe N8N fonctionne (cycle = "Achats" est détecté)

❌ PROBLÈME : Le node "Code integrate7" n'envoie PAS les autres variables
   (user_message, constat, anomalie)

🎯 CAUSE RACINE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Le node "Code integrate7" transforme les données du webhook mais ne crée PAS
les champs user_message, constat, anomalie dans son output.

Il crée seulement :
{
  "data": [
    { "User_message": [...] },
    { "filename.xlsx": [...] }
  ]
}

Mais l'agent RAG attend :
{
  "user_message": "...",
  "cycle": "...",
  "constat": "...",
  "anomalie": "..."
}

✅ SOLUTION 1 : MODIFIER LE NODE "CODE INTEGRATE7"
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Ajouter à la fin du code JavaScript du node "Code integrate7" :

// ============================================================================
// EXTRACTION DES VARIABLES POUR L'AGENT RAG
// ============================================================================

let extractedMessage = "";
let extractedCycle = "Achats";
let extractedConstat = "";
let extractedAnomalie = "";

// Extraire user_message depuis User_message array
if (result.length > 0 && result[0].User_message) {
    const userMsgArray = result[0].User_message;
    
    for (const item of userMsgArray) {
        const key = Object.keys(item)[0];
        const value = item[key];
        
        if (key.includes('Message') || key.includes('message')) {
            extractedMessage += value + " ";
        }
        if (key.includes('Cycle') || key.includes('cycle')) {
            extractedCycle = value;
        }
        if (key.includes('Constat') || key.includes('constat')) {
            extractedConstat = value;
        }
        if (key.includes('Anomalie') || key.includes('anomalie')) {
            extractedAnomalie = value;
        }
    }
}

// Retourner avec les variables extraites
return [{
    json: {
        data: result,
        user_message: extractedMessage.trim() || "Message non fourni",
        cycle: extractedCycle,
        constat: extractedConstat || "À compléter",
        anomalie: extractedAnomalie || "À compléter"
    }
}];

✅ SOLUTION 3 : AJOUTER UN NODE "CODE" APRÈS "CODE INTEGRATE7"
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

RECOMMANDÉ : Ajouter un nouveau node "Code" entre "Code integrate7" et "Alea16"

Nom du node : "Extract Variables for RAG"

Code JavaScript :

const inputData = $input.first().json;
const result = {
    data: inputData.data,
    user_message: "Message non fourni",
    cycle: "Achats",
    constat: "À compléter",
    anomalie: "À compléter"
};

// Extraire les variables depuis User_message
if (inputData.data && inputData.data.length > 0) {
    const userMessageObj = inputData.data.find(item => item.User_message);
    
    if (userMessageObj && userMessageObj.User_message) {
        const userMsgArray = userMessageObj.User_message;
        
        for (const item of userMsgArray) {
            const key = Object.keys(item)[0];
            const value = item[key];
            
            // Extraire selon le nom de la clé
            if (key.toLowerCase().includes('message')) {
                result.user_message = value;
            } else if (key.toLowerCase().includes('cycle')) {
                result.cycle = value;
            } else if (key.toLowerCase().includes('constat')) {
                result.constat = value;
            } else if (key.toLowerCase().includes('anomalie')) {
                result.anomalie = value;
            }
        }
    }
}

return [{ json: result }];
