// Créer un utilisateur de test const testUser = { id: 'test-user-' + Date.now(), name: 'Utilisateur Test', email: 'test@example.com', password: 'test123', createdAt: new Date(), lastLogin: new Date(), personalInfo: { name: 'Utilisateur Test', email: 'test@example.com', company: 'Société Test', role: 'Auditeur', theme_preference: 'system' } }; // Sauvegarder dans IndexedDB db.users.add(testUser).then(() => { return db.setCurrentUser(testUser.id); }).then(() => { console.log('✅ Utilisateur de test créé et connecté'); console.log('🔄 Rechargement de la page...'); setTimeout(() => location.reload(), 500); }).catch(error => { console.error('❌ Erreur:', error); });
