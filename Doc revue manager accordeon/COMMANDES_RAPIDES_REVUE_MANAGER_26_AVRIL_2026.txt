📊 STRUCTURE DES DONNÉES
────────────────────────

# Format RevuePoint
{
  "num": 1,
  "titre": "Point de revue",
  "statut": "ok",  // "ok" | "warn" | "ko"
  "prep": "Commentaire préparateur",
  "rev": "Commentaire superviseur"
}

# Clé localStorage
clara_revue_manager_${reference}

# Exemples de clés
clara_revue_manager_Test-Caisse-001
clara_revue_manager_REF-AUDIT-2023
clara_revue_manager_PV-C-23-009

🎨 STATUTS DISPONIBLES
──────────────────────

ok   → Validé    (vert)   #EAF3DE / #3B6D11
warn → À suivre  (ambre)  #FAEEDA / #854F0B
ko   → À ajuster (rouge)  #FCEBEB / #A32D2D

🔍 DÉTECTION AUTOMATIQUE
────────────────────────

# Mots-clés pour "ok"
val, ok, satisf

# Mots-clés pour "warn"
suivre, warn, attention

# Mots-clés pour "ko"
ajust, ko, anomal

# Fallback
Contenu présent → "ok"
Pas de contenu → "warn"

# Exporter vers JSON
const exportToJson = () => {
  const data = {};
  Object.keys(localStorage)
    .filter(k => k.startsWith('clara_revue_manager_'))
    .forEach(k => data[k] = JSON.parse(localStorage.getItem(k)));
  
  const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'revue_manager_export.json';
  a.click();
};

# Importer depuis JSON
const importFromJson = (jsonString) => {
  const data = JSON.parse(jsonString);
  Object.keys(data).forEach(k => {
    localStorage.setItem(k, JSON.stringify(data[k]));
  });
  console.log('Import terminé:', Object.keys(data).length, 'clés');
};
