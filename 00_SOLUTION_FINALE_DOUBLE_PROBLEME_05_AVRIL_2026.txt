```python
def remplir_onglet_dynamique(ws, postes, col_debut, ligne_debut):
    """
    Remplit un onglet dynamiquement sans mapping fixe
    
    Args:
        ws: Worksheet
        postes: Liste des postes [{ref, libelle, montant_n, montant_n1}, ...]
        col_debut: Colonne de début (ex: 'C')
        ligne_debut: Ligne de début (ex: 10)
    """
    ligne = ligne_debut
    
    for poste in postes:
        # Écrire montant N
        ws[f'{col_debut}{ligne}'] = poste.get('montant_n', 0)
        
        # Écrire montant N-1 (colonne suivante)
        col_n1 = chr(ord(col_debut) + 1)
        ws[f'{col_n1}{ligne}'] = poste.get('montant_n1', 0)
        
        ligne += 1
```
