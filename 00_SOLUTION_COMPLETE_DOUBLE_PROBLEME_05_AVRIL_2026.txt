ÉTAPE 1: REMPLIR L'ONGLET BILAN
--------------------------------

L'onglet BILAN est le MASTER qui alimente tous les autres onglets.

Cellules à remplir dans BILAN:
- Colonne E (lignes 11-39): Montants Exercice N
- Colonne F (lignes 11-39): Montants Exercice N-1
- Colonne G (lignes 11-39): Montants Exercice N-2

Structure:
- Lignes 11-26: ACTIF
- Lignes 27-39: PASSIF

```python
def generer_etats_controle_pour_export(results: Dict[str, Any]) -> List[Dict[str, Any]]:
    """
    Génère les 16 états de contrôle à partir des résultats des états financiers
    UTILISE LES 16 ÉTATS ACTUALISÉS de etats_controle_exhaustifs.py
    """
    from etats_controle_exhaustifs import generer_controles_exhaustifs
    
    # Extraire les données
    balance_n = results.get('balance_n', [])
    balance_n1 = results.get('balance_n1', [])
    bilan_actif = results.get('bilan_actif', [])
    bilan_passif = results.get('bilan_passif', [])
    compte_resultat = results.get('compte_resultat', [])
    
    # Générer les contrôles exhaustifs
    controles_n, controles_n1, totaux_n, totaux_n1 = generer_controles_exhaustifs(
        balance_n=balance_n,
        balance_n1=balance_n1,
        bilan_actif=bilan_actif,
        bilan_passif=bilan_passif,
        compte_resultat=compte_resultat
    )
    
    # Convertir en format Excel
    etats_controle = []
    
    # 1. Statistiques de Couverture (N)
    stats = controles_n.get('statistiques', {})
    etats_controle.append({
        'titre': '1. Statistiques de Couverture (Exercice N)',
        'postes': [
            {'ref': 'CA', 'libelle': 'Comptes dans la balance', 'montant_n': stats.get('total_comptes_balance', 0), 'montant_n1': 0},
            {'ref': 'CB', 'libelle': 'Comptes intégrés', 'montant_n': stats.get('comptes_integres', 0), 'montant_n1': 0},
            {'ref': 'CC', 'libelle': 'Comptes non intégrés', 'montant_n': stats.get('comptes_non_integres', 0), 'montant_n1': 0},
            {'ref': 'CD', 'libelle': 'Taux de couverture (%)', 'montant_n': stats.get('taux_couverture', 0), 'montant_n1': 0},
        ]
    })
    
    # 2. Équilibre du Bilan (N)
    equilibre = controles_n.get('equilibre_bilan', {})
    etats_controle.append({
        'titre': '2. Équilibre du Bilan (Exercice N)',
        'postes': [
            {'ref': 'EA', 'libelle': 'Total Actif', 'montant_n': equilibre.get('actif', 0), 'montant_n1': 0},
            {'ref': 'EB', 'libelle': 'Total Passif', 'montant_n': equilibre.get('passif', 0), 'montant_n1': 0},
            {'ref': 'EC', 'libelle': 'Différence (Actif - Passif)', 'montant_n': equilibre.get('difference', 0), 'montant_n1': 0},
            {'ref': 'ED', 'libelle': 'Pourcentage d\'écart (%)', 'montant_n': equilibre.get('pourcentage_ecart', 0), 'montant_n1': 0},
        ]
    })
    
    # ... Continuer pour les 14 autres états
    
    return etats_controle
```
