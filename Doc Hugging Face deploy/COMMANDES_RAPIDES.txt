═══════════════════════════════════════════════════════════════════════════════
   🚀 DÉPLOIEMENT AUTOMATIQUE (RECOMMANDÉ)
═══════════════════════════════════════════════════════════════════════════════

# 1. Aller dans le dossier py_backend
cd py_backend

# 2. Exécuter le script de déploiement
..\Doc Hugging Face deploy\deploy-huggingface.ps1

# 3. Suivre les instructions à l'écran

═══════════════════════════════════════════════════════════════════════════════
   📝 DÉPLOIEMENT MANUEL
═══════════════════════════════════════════════════════════════════════════════

# 1. Aller dans le dossier py_backend
cd py_backend

# 2. Initialiser Git (si pas déjà fait)
git init

# 3. Ajouter le remote Hugging Face
# Remplacer [USERNAME] par votre nom d'utilisateur HF
git remote add hf https://huggingface.co/spaces/[USERNAME]/claraverse-backend

# 4. Configurer Git
git config user.name "Votre Nom"
git config user.email "votre@email.com"

# 5. Ajouter tous les fichiers
git add .

# 6. Créer un commit
git commit -m "Initial deployment: Claraverse backend"

# 7. Pousser vers Hugging Face
git push hf main

# Entrer vos credentials quand demandé :
# Username: [votre-username-huggingface]
# Password: [votre-token-huggingface]

═══════════════════════════════════════════════════════════════════════════════
   🔄 MISE À JOUR DU DÉPLOIEMENT
═══════════════════════════════════════════════════════════════════════════════

# Après avoir fait des modifications :

# 1. Aller dans py_backend
cd py_backend

# 2. Ajouter les modifications
git add .

# 3. Créer un commit
git commit -m "Update: description des changements"

# 4. Pousser vers Hugging Face
git push hf main

═══════════════════════════════════════════════════════════════════════════════
   ✅ VÉRIFICATION
═══════════════════════════════════════════════════════════════════════════════

# Tester l'endpoint de santé (remplacer [USERNAME])
curl https://[USERNAME]-claraverse-backend.hf.space/health

# Ou dans PowerShell :
Invoke-WebRequest -Uri "https://[USERNAME]-claraverse-backend.hf.space/health"

# Ouvrir la documentation dans le navigateur
start https://[USERNAME]-claraverse-backend.hf.space/docs

═══════════════════════════════════════════════════════════════════════════════
   🔧 COMMANDES UTILES
═══════════════════════════════════════════════════════════════════════════════

# Vérifier la version de Git
git --version

# Vérifier les remotes configurés
git remote -v

# Voir le statut Git
git status

# Voir l'historique des commits
git log --oneline

# Supprimer le remote Hugging Face (si besoin de reconfigurer)
git remote remove hf

# Forcer le push (attention : écrase l'historique distant)
git push hf main --force

═══════════════════════════════════════════════════════════════════════════════
   🐳 TESTS LOCAUX AVEC DOCKER
═══════════════════════════════════════════════════════════════════════════════

# Construire l'image Docker localement
docker build -t claraverse-backend .

# Lancer le container
docker run -p 7860:7860 claraverse-backend

# Tester l'API localement
curl http://localhost:7860/health

# Ouvrir la documentation locale
start http://localhost:7860/docs

# Arrêter le container
docker ps  # Noter le CONTAINER ID
docker stop [CONTAINER_ID]

git config --global credential.helper wincred

# Ou utiliser le store (stocke en clair - attention)
git config --global credential.helper store

# Supprimer les credentials stockés
git credential-wincred erase

# Créer un alias pour le déploiement rapide
Set-Alias -Name deploy-hf -Value "..\Doc Hugging Face deploy\deploy-huggingface.ps1"

# Utiliser ensuite simplement :
deploy-hf

# Créer une fonction PowerShell pour tester rapidement
function Test-HFApi {
    param([string]$Username)
    $url = "https://$Username-claraverse-backend.hf.space/health"
    Invoke-WebRequest -Uri $url | Select-Object StatusCode, Content
}

# Utiliser :
Test-HFApi -Username "votre-username"
