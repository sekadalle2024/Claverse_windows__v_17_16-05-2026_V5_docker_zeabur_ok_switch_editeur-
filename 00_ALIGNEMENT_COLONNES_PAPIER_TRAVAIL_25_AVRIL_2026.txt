═══════════════════════════════════════════════════════════════════════════════

emptyColumnsCount = ecartIndex - nbVariables + 1
remainingColumns = totalColumns - emptyColumnsCount - nbVariables

- remainingColumns = 6 - 2 - 3 = 1
